**adbc_scanner.h:**
```cpp
#pragma once

#include <arrow-adbc/adbc.h>
#include <arrow/record_batch.h>
#include <memory>
#include <string>

#include "column/chunk.h"
#include "common/status.h"
#include "runtime/runtime_state.h"

namespace starrocks {

class ADBCScanner {
public:
    ADBCScanner(std::string driver, std::string uri, std::string username, std::string password,
                std::string token, std::string sql, const TupleDescriptor* tuple_desc);
    ~ADBCScanner();

    Status open(RuntimeState* state);
    Status get_next(RuntimeState* state, ChunkPtr* chunk, bool* eos);
    void close(RuntimeState* state);

    int64_t rows_read() const { return _rows_read; }
    int64_t bytes_read() const { return _bytes_read; }

private:
    Status _init_adbc();
    Status _convert_batch_to_chunk(const std::shared_ptr<arrow::RecordBatch>& batch, ChunkPtr* chunk);

    // Connection parameters
    std::string _driver;
    std::string _uri;
    std::string _username;
    std::string _password;
    std::string _token;
    std::string _sql;
    const TupleDescriptor* _tuple_desc;

    // ADBC C API handles (must be released in reverse order)
    AdbcDatabase _database{};
    AdbcConnection _connection{};
    AdbcStatement _statement{};
    bool _database_initialized = false;
    bool _connection_initialized = false;
    bool _statement_initialized = false;

    // Arrow reader
    std::shared_ptr<arrow::RecordBatchReader> _batch_reader;

    // Re-chunking state
    std::shared_ptr<arrow::RecordBatch> _pending_batch;
    int64_t _pending_offset = 0;
    int64_t _max_chunk_size = 4096;

    // Stats
    int64_t _rows_read = 0;
    int64_t _bytes_read = 0;
    bool _opened = false;
    bool _closed = false;
};

} // namespace starrocks
```

**adbc_scanner.cpp:**

1. Define `RETURN_ADBC_NOT_OK(status_code, adbc_error)` macro that:
   - Checks if `status_code != ADBC_STATUS_OK`
   - Extracts error.message (or "Unknown ADBC error" if null)
   - Calls `error.release(&error)` if release function is set
   - Returns `Status::InternalError("ADBC error: " + msg)`

2. **Constructor:** Store all parameters. Set `_max_chunk_size` from RuntimeState config (state->chunk_size()) in open(), or default 4096.

3. **open():**
   - Call `_init_adbc()` to set up database/connection/statement
   - Execute query: `AdbcStatementExecuteQuery(&_statement, &c_stream, &rows_affected, &error)`
   - Import stream: `auto result = arrow::ImportRecordBatchReader(&c_stream)`
   - Check result status, store in `_batch_reader`
   - Set `_opened = true`

4. **_init_adbc():**
   - `AdbcDatabaseNew(&_database, &error)` + RETURN_ADBC_NOT_OK
   - `AdbcDatabaseSetOption(&_database, "driver", _driver.c_str(), &error)` -- use the driver name directly (e.g., "adbc_driver_flightsql"). The FE translates "flight_sql" to "adbc_driver_flightsql" in ADBCScanNode.
   - `AdbcDatabaseSetOption(&_database, "uri", _uri.c_str(), &error)`
   - If `_username` not empty: `AdbcDatabaseSetOption(&_database, "username", ...)`
   - If `_password` not empty: `AdbcDatabaseSetOption(&_database, "password", ...)`
   - If `_token` not empty: `AdbcDatabaseSetOption(&_database, "adbc.flight.sql.authorization_header", ("Bearer " + _token).c_str(), ...)`
   - `AdbcDatabaseInit(&_database, &error)` + RETURN_ADBC_NOT_OK, set `_database_initialized = true`
   - `AdbcConnectionNew(&_connection, &error)` + RETURN_ADBC_NOT_OK
   - `AdbcConnectionInit(&_connection, &_database, &error)` + RETURN_ADBC_NOT_OK, set `_connection_initialized = true`
   - `AdbcStatementNew(&_connection, &_statement, &error)` + RETURN_ADBC_NOT_OK, set `_statement_initialized = true`
   - `AdbcStatementSetSqlQuery(&_statement, _sql.c_str(), &error)` + RETURN_ADBC_NOT_OK

5. **get_next():**
   - If `_pending_batch` has remaining rows (from re-chunking), consume from `_pending_offset`
   - Otherwise call `_batch_reader->ReadNext(&batch)`. If batch is null, set `*eos = true`, return OK.
   - If batch->num_rows() > `_max_chunk_size`: store as `_pending_batch`, set `_pending_offset = 0`, slice to `_max_chunk_size` rows using `batch->Slice(offset, length)`
   - Call `_convert_batch_to_chunk(sliced_batch, chunk)`
   - Update `_rows_read` and `_bytes_read` stats

6. **_convert_batch_to_chunk():**
   - Create a new Chunk with columns matching `_tuple_desc->slots()`
   - For each materialized slot in `_tuple_desc`:
     - Get arrow column from batch by index
     - Use `get_arrow_converter()` from `arrow_to_starrocks_converter.h` to get ConvertFunc
     - Call converter to fill StarRocks Column
   - Set chunk num_rows
   - Return OK

7. **close():**
   - Guard against double-close: if `_closed` return
   - Release in reverse order: statement, connection, database
   - Only release if the corresponding `_initialized` flag is true
   - Use AdbcError for each release call but don't fail on release errors (log warning)
   - Set `_closed = true`

8. **Destructor:** Call close(nullptr) if not already closed.

CRITICAL PITFALLS (from research):
- Hold AdbcStatement/Connection/Database alive until RecordBatchReader is fully consumed (they are released in close(), not after execute)
- Always check ADBC_STATUS_OK on every C API call
- Use RAII-like flags (_database_initialized etc.) to ensure correct cleanup even on partial initialization failure
- The driver name passed to AdbcDatabaseSetOption must be "adbc_driver_flightsql" (the C library name), NOT "flight_sql" (the user-facing property). The FE performs this translation.

Include headers: `<arrow-adbc/adbc.h>`, `<arrow/c/bridge.h>`, `<arrow/record_batch.h>`, `"exec/arrow_to_starrocks_converter.h"`, `"column/chunk.h"`, `"runtime/descriptors.h"`

class ADBCDataSource;
class ADBCDataSourceProvider final : public DataSourceProvider {
public:
    friend class ADBCDataSource;
    ADBCDataSourceProvider(ConnectorScanNode* scan_node, const TPlanNode& plan_node);
    ~ADBCDataSourceProvider() override = default;
    DataSourcePtr create_data_source(const TScanRange& scan_range) override;
    bool insert_local_exchange_operator() const override { return true; }
    bool accept_empty_scan_ranges() const override { return false; }
    const TupleDescriptor* tuple_descriptor(RuntimeState* state) const override;

protected:
    ConnectorScanNode* _scan_node;
    TADBCScanNode _adbc_scan_node;

class ADBCDataSource final : public DataSource {
public:
    ADBCDataSource(const ADBCDataSourceProvider* provider, const TScanRange& scan_range);
    ~ADBCDataSource() override = default;
    std::string name() const override { return "ADBCDataSource"; }
    Status open(RuntimeState* state) override;
    void close(RuntimeState* state) override;
    Status get_next(RuntimeState* state, ChunkPtr* chunk) override;
    int64_t raw_rows_read() const override { return _rows_read; }
    int64_t num_rows_read() const override { return _rows_read; }
    int64_t num_bytes_read() const override { return _bytes_read; }
    int64_t cpu_time_spent() const override { return 0; }

private:
    const ADBCDataSourceProvider* _provider;
    std::unique_ptr<ADBCScanner> _scanner;
    int64_t _rows_read = 0;
    int64_t _bytes_read = 0;

};

} // namespace starrocks
